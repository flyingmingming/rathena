-	script	hourly_points	-1,{

OnHourlyRewards:

    if(vip_status(VIP_STATUS_ACTIVE)){
        addtimer .reward_time, strnpcinfo(3)+"::OnHourlyRewards";
        .reward = 2;
        callsub Hourly_Rewards;
        end;
    }

    if(checkvending()){
        message strcharinfo(0),"[Hourly Rewards]: No Hourly Rewards for Venders , Relogin to change that .";
        end;
    }

    callsub Hourly_Rewards;

end;

OnPCLogoutEvent:

end;
 
OnPCLoginEvent:
	addtimer .reward_time, strnpcinfo(3)+"::OnHourlyRewards";
end;

OnInit:
    .idle_time = 1800;              //	this the time for idle in secend  , DEFAULT = 1800 (30 min)
    .vip = 0;                       //	1 = only for VIP / 0 = for everyone  , DEFAULT = 0
    .reward_time = 1000;            //	1000 = 1 secend | 60000 = 1 min | 3600000 = 1 houre  , DEFAULT = 3600000 (1 hr)
    .reward = 1;                    //  hourly points reward
end;

Hourly_Rewards:

	#ascendance_points += .reward;            //put the Hourly Rewards here
	message strcharinfo(0),"You have gained " + .reward + " hourly points. Total: "+F_InsertComma(#ascendance_points)+".";

return;

}
